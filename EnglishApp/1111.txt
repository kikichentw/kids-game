    function shuffleArray(array) {
        let currentIndex = array.length, randomIndex;
        while (currentIndex !== 0) {
            randomIndex = Math.floor(Math.random() * currentIndex);
            currentIndex--;
            [array[currentIndex], array[randomIndex]] = [array[randomIndex], array[currentIndex]];
        }
        return array;
    }

    function playBeep() {
        try {
            const ctx = new (window.AudioContext || window.webkitAudioContext)();
            const osc = ctx.createOscillator();
            osc.type = 'sine';
            osc.frequency.setValueAtTime(880, ctx.currentTime);
            osc.connect(ctx.destination);
            osc.start();
            osc.stop(ctx.currentTime + 0.15); 
        } catch(e) {}
    }

    function playAudio(text, lang = 'en-US', rate = 0.4) {
        try {
            if ('speechSynthesis' in window) {
                window.speechSynthesis.cancel();
                let msg = new SpeechSynthesisUtterance(text);
                msg.lang = lang;
                msg.rate = rate;
                window.speechSynthesis.speak(msg);
            }
        } catch(e) {}
    }

    function unlockAudio() {
        if (!audioUnlocked) {
            try {
                if ('speechSynthesis' in window) {
                    let unlockMsg = new SpeechSynthesisUtterance('');
                    window.speechSynthesis.speak(unlockMsg);
                }
                const winSound = document.getElementById('win-sound');
                if(winSound) winSound.load(); 
            } catch(e) {}
            audioUnlocked = true;
        }
    }

    function loadMC() {
        // 💡 條件改成只要 >= 5 就觸發勝利動畫
        if(currentStep >= 5) return triggerWin(); 
        
        const q = currentQuestionPool[currentStep]; 
        let options = [q]; 
        while(options.length < 4) {
            let randomOpt = vocabData[Math.floor(Math.random() * vocabData.length)];
            if(!options.some(opt => opt.en === randomOpt.en)) options.push(randomOpt);
        }
        options = shuffleArray(options); 

        // 💡 進度條也改成顯示 / 5
        let html = `
            <div class="progress-text">🌟 ${currentStep + 1} / 5 🌟</div>
            <div class="question-word" onclick="playAudio('${q.en}', 'en-US', 0.4)">🔊 ${q.en}</div>
        `;
        
        options.forEach(opt => {
            html += `
                <div class="option-container">
                    <div class="option-content" onclick="checkMC('${opt.en}', '${q.en}')">
                        <span class="option-icon">${opt.ic}</span>
                        <div class="option-text-group">
                            <span class="option-meaning">${opt.zh}</span>
                            <span class="option-zhuyin">${opt.zy}</span>
                        </div>
                    </div>
                    <div class="speaker-icon" onclick="playAudio('${opt.zh}', 'zh-TW', 0.8)">🔊</div>
                </div>
            `;
        });
        document.getElementById('game-content').innerHTML = html;
        setTimeout(() => playAudio(q.en, 'en-US', 0.4), 300);
    }

    function checkMC(selectedEn, correctEn) {
        if(selectedEn === correctEn) {
            currentStep++; playAudio(selectedEn, 'en-US', 0.4); setTimeout(loadMC, 600); 
        } else { 
            showWrongGif(); 
        }
    }

    function loadMatch() {
        if(currentStep >= currentQuestionPool.length) return triggerWin(); 
        matchCount = 0;
        let words = currentQuestionPool.slice(currentStep, currentStep + 5); 
        
        let enList = shuffleArray([...words]);
        let zhList = shuffleArray([...words]);

        while(enList.every((val, index) => val.en === zhList[index].en) && words.length > 1) {
            zhList = shuffleArray([...words]);
        }

        let roundNumber = Math.floor(currentStep / 5) + 1; 
        let html = `
            <div class="progress-text">🌟 第 ${roundNumber} / 6 關 🌟</div>
            <div class="match-container">
                <div class="match-col" id="col-en">
                    ${enList.map(item => `<div class="match-btn" id="en-${item.en}" onclick="selectMatch('en', '${item.en}')">${item.ic} ${item.en}</div>`).join('')}
                </div>
                <div class="match-col" id="col-zh">
                    ${zhList.map(item => `<div class="match-btn" id="zh-${item.en}" onclick="selectMatch('zh', '${item.en}')">${item.zh}</div>`).join('')}
                </div>
            </div>
        `;
        document.getElementById('game-content').innerHTML = html;
    }

    function selectMatch(lang, wordEn) {
        if (lang === 'en') playAudio(wordEn, 'en-US', 0.4);
        else {
            const wordData = vocabData.find(item => item.en === wordEn);
            playAudio(wordData.zh, 'zh-TW', 0.8);
        }

        let btn = document.getElementById(lang + '-' + wordEn);
        document.querySelectorAll(`#col-${lang} .match-btn`).forEach(b => b.classList.remove('selected'));
        btn.classList.add('selected');
        matchSelection[lang] = wordEn;

        if(matchSelection.en && matchSelection.zh) {
            if(matchSelection.en === matchSelection.zh) {
                document.getElementById('en-' + matchSelection.en).classList.add('matched');
                document.getElementById('zh-' + matchSelection.zh).classList.add('matched');
                matchCount++;
                if(matchCount === 5) {
                    currentStep += 5;
                    setTimeout(loadMatch, 800);
                }
            } else {
                showWrongGif(); 
                setTimeout(() => {
                    document.getElementById('en-' + matchSelection.en).classList.remove('selected');
                    document.getElementById('zh-' + matchSelection.zh).classList.remove('selected');
                }, 400);
            }
            matchSelection = { en: null, zh: null };
        }
